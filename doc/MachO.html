def self.open(filename)
	# open file and test magic instead of using exceptions for control?
	begin
		file = MachOFile.new(filename)
	rescue FatBinaryError
		file = FatFile.new(filename)
	end

	file
end

def self.round(value, round)
	round -= 1
	value += round
	value &= ~round
	value
end
